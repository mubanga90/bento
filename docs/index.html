and this bit of JavaScript:

            import Bento from '../dist/main.js';

const bentoOptions = {
    preferredCellSize: { x: 47, y: 33 },
};
                
const gridElement = document.querySelector('#grid');
const _bentoInstance = new Bento(gridElement, bentoOptions);
